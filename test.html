function initialize() { var chicago = new google.maps.LatLng(41.399095, -73.44344);//41.875696,-87.624207); var mapOptions = { zoom: 11, center: chicago } var map = new google.maps.Map(document.getElementById('map-canvas'), mapOptions); var ctaLayer = new google.maps.KmlLayer({ url: 'http://crkrav.herokuapp.com/bingroute.kml' }); ctaLayer.setMap(map); } google.maps.event.addDomListener(window, 'load', initialize);
